Spring注解

Xuxx 2019-12-30 in 笔记

1.属性注入

1.@Value

实体类

@Component
public class Book {

    @Value("${book.id}")
    private Long id;
    @Value("${book.name}")
    private String name;
    @Value("${book.author}")
    private String author;
    //get/set/toString方法略

配置文件

book.id=1
book.name=Java开发
book.author=Xuxx

测试

@Autowired
    Book book;

    @Test
    void contextLoads() {
        System.out.println(book);
    }
2. @PropertySource

创建配置文件book.properties

book.id=1
book.name=Java开发2.0
book.author=Xuxx

实体类

@Component
@PropertySource(value = "classpath:book.properties")
public class Book {

    @Value("${book.id}")
    private Long id;
    @Value("${book.name}")
    private String name;
    @Value("${book.author}")
    private String author;
    //get/set/toString方法略

测试

@Autowired
    Book book;

    @Test
    void contextLoads() {
        System.out.println(book);
    }

book.id=1

book.author=Xuxx

@Component
@PropertySource(value = "classpath:book.properties")
@ConfigurationProperties(prefix = "book")
public class Book {
    private Long id;
    private String name;
    private String author;
    //get/set/toString方法略

测试

    @Test

    }
